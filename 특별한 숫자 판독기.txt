#include <stdio.h>

int main()
{
    int a, x, y, z, first, last;
    
    x = (a / 100) % 10;
    y = (a / 10) % 10;
    z = a % 10;

    printf("input number 0 to 999\n");
    scanf("%d", &a);
        if (a < 0 || a > 999) {
            printf("input 3-digit number");
            return 0;
    }
    if (a % 3 == 0 && a % 2 == 1) {
        printf("Special number!");
        return 0;
    } else {
        printf("Normal number...");
        return 0;
    }
    if ((x + y + z) % 7 == 0) {
        printf("Special number!");
        return 0;
    } else {
        printf("Normal number...");
        return 0;
    }
    if (a >= 100) {
        first = x;
    } else if (a >= 10) {
        first = y;
    } else {
        first = z;
    }
    last = z;
    
    if (first > last) {
        printf("Special number!");
        return 0;
    } else {
        printf("Normal number...");
    }
    return 0;
}
