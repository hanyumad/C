        printf("input 3-Digit number");
        return 0;
    } else if (M > N) {
        printf("M은 N보다 클 수 없습니다");
        return 0;
    }
    
    for (i = M; i < N; i++) {
        if (i % 7 == 0 ||  i % 10 == 7 || i % 100 == 7) {
        } else if (i % 5 == 0) {
            printf("%d\n", i);
            continue;
        } else {
            printf("%d\n", i);
        }
    }

}
