#include <stdio.h>

int main() {

    float a = 0;
    
    printf("input score\n");
    scanf("%f", &a);
    
    if (a < 0 || a > 100) {
        printf("input value score");
        return -1;
    }

    switch((int)a / 10) {
        case 10:// 100
        case 9: //  90~99
            printf("grade A\n");
            break;
        case 8: //  80~89
            printf("grade B\n");
            break;
        case 7: //  70~79
            printf("grade C\n");
            break;
        case 6: //  60~69
            printf("grade D\n");
            break;
        default://  60미만
            printf("grade F\n");
            break;
    }

    if (a >= 90 && a <= 100) {
        a = a + (a * 0.1);
        } 
    else if (a >= 80 && a < 90) {
        a = a + (a * 0.05);
    }

    printf("total score is %.2f", a);
    return 0;
}
