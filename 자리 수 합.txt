#include <stdio.h>

int main()
{
    int i, n, sum = 0;
    printf("세자리 수를 입력하세요\n");
    scanf("%d", &n);
    
    for (i = n; i > 0; i = i / 10) {
        sum = sum + i % 10;
        if (n >= 1000) {
            printf("세자리 수를 입력하세요");
            return 0;
        }
    }
    printf("자리 수의 합 : %d", sum);
    return 0;
}
