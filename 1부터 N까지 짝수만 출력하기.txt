#include <stdio.h>

int main() {
    
    int n, i = 1;
    
    printf("input number\n");
    scanf("%d", &n);
    
    if (n <= 0) {
        printf("input value number");
        return -1;
    }

    while (i <= n) {
        if (i % 2 == 1) {  
            i = i + 1;
            continue;
        }
        printf("%d ", i);
        i = i + 1;
    }
    return 0;
}
