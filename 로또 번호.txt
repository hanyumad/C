#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define SIZE 6

void SelectionSort(int sum[]);

int main() {
    
    int num[SIZE];
    int i, j;
    
    srand(time(NULL));
    
    for (i = 0; i < SIZE; i++) {
        num[i] = (int)(rand() % 45) + 1;
        for (j = 0; j < i; j++) {
            if(num[i] == num[j]) i--;
        }
    }
    
    SelectionSort(num);
    
        for (i = 0; i < SIZE; i++) {    
            printf("%d\n", num[i]);
            sleep(2);
    } 
    return 0;
}

void SelectionSort(int sum[]) {
    int i, j, min, temp;
    
    for(i = 0; i < SIZE - 1; i++) {
        min = i;
        for(j = i + 1; j < SIZE; j++) {
            if(sum[j] < sum[min]) min = j;
        }
        temp = sum[i];
        sum[i] = sum[min];
        sum[min] = temp;
    }
}
