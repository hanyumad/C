#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define SIZE 6

int main() {
    
    int num[SIZE];
    int i, j;

    srand(time(NULL));

    // 중복 없는 난수 생성
    for (i = 0; i < SIZE; i++) {
        num[i] = (rand() % 45) + 1;
        for (j = 0; j < i; j++) {
            if (num[i] == num[j]) {
                i--;  // 중복이면 다시 뽑기
                break;
            }
        }
    }

    // 선택 정렬
    for (i = 0; i < SIZE - 1; i++) {
        int min = i;
        for (j = i + 1; j < SIZE; j++) {
            if (num[j] < num[min]) {
                min = j;
            }
        }
        // 교환
        int temp = num[i];
        num[i] = num[min];
        num[min] = temp;
    }

    // 출력
    for (i = 0; i < SIZE; i++) {
        printf("%d\n", num[i]);
        //sleep(1);
    }

    return 0;
}
