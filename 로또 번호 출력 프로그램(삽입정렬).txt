#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define SIZE 6

int main() {
    
    int num[SIZE];
    int i, j;

    srand(time(NULL));  // rand 함수의 시드 설정
                        // time(NULL)은 현재 시간을 가져옴. 이 값으로 난수 생성
    // 중복 없는 난수 생성
    for (i = 0; i < SIZE; i++) {    // 0부터 5까지 총 6번 반복
        num[i] = (rand() % 45) + 1;
        for (j = 0; j < i; j++) {
            if (num[i] == num[j]) {
                i--;  // 중복이면 다시 뽑기
                break;
            }
        }
    }
    // 삽입정렬
    int temp;
    
    for (int i = 1; i < SIZE; i++) { 
    // 자신보다 앞의 요소와 비교하기 때문에 1부터 시작
        int j = i - 1; // 앞의 요소
        while((num[j] > num[j + 1]) && (j >= 0)) {
        // 현재 값이 그 앞의 값보다 작고, 0 이상일 때 반복
        // 정렬된 부분에서 현재 값보다 큰 값들을 오른쪽으로 이동
            temp = num[j];
            num[j] = num[j + 1];
            num[j + 1] = temp; // 값을 교환하며 정렬
            j--; // 비교 위치를 한칸 왼쪽으로 이동
        }
    }
    // 출력
    for (i = 0; i < SIZE; i++) {
        printf("%d\n", num[i]);
        //sleep(1);
    }
    return 0;
}
