#include <stdio.h>

int main()
{
    int result, a, b, c = 0;

    scanf("%d", &a);
    scanf("%d", &b);
    scanf("%d", &c);
    
    if (c <= 0) {
        printf("1 이상의 숫자를 입력하세요.");
        return 0;
    }
    result = (a + b) * c - (a - b) / c;
    printf("%d", result);
    return 0;
}
