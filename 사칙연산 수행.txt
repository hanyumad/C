#include <stdio.h>

int main()
{
    int result = 0;
    int a = 0;
    int b = 0;
    int c = 0;
    
    scanf("%d", &a);
    scanf("%d", &b);
    scanf("%d", &c);
    
    if (c == 0 || c <= -1) {    // c가 0이거나  -1보다 작다면
        printf("1 이상의 숫자를 입력하세요");
        return 0;
    }
    result = (a + b) * c - (a - b) / c;
    printf("%d", result);
    return 0;
}
